## Overview

This method is responsible for returning a chat's metadata information.

---

## Attributes

### Header

<InstanceId />
<Token />


<ParamField header="Client-Token" type="string">
  Account security token
</ParamField>

### Path

<ParamField path="phone" type="string" required>
  Contact phone number in DDI DDD NUMBER format. E.g.: 551199999999
</ParamField>

---

## Response

### 200

<ResponseField name="phone" type="string">
  Contact phone number
</ResponseField>

<ResponseField name="unread" type="string">
  Number of unread messages
</ResponseField>

<ResponseField name="lastMessageTime" type="string">
  Last interaction timestamp
</ResponseField>

<ResponseField name="isMuted" type="string">
  0 or 1 indicating if the chat is muted
</ResponseField>

<ResponseField name="isMarkedSpam" type="boolean">
  Whether the chat is marked as spam
</ResponseField>

<ResponseField name="profileThumbnail" type="string">
  Chat photo URL (deleted after 48h by WhatsApp)
</ResponseField>

<ResponseField name="isGroupAnnouncement" type="boolean">
  Whether it is an announcement group
</ResponseField>

<ResponseField name="isGroup" type="boolean">
  Whether it is a group
</ResponseField>

<ResponseField name="about" type="string">
  Profile status
</ResponseField>

<ResponseField name="notes" type="object">
  Chat annotations (available only for WhatsApp Business)
</ResponseField>
